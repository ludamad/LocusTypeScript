!!! A 'continue' statement can only be used within an enclosing iteration statement.
    
    // non-existent label
    ONE:
    do continue TWO; while (true)
    
    // continue from inside function
    TWO:

            continue TWO;
        }
    }while (true)
    
    THREE:
    do {
        var fn = function () {
            continue THREE;
        }
    }while (true)
    
    // continue forward

        continue FIVE;
        FIVE:
        do { } while (true)
    }while (true)
